[PHP] switch文の乱用

投稿者からのアピールポイント

臭うファイルがあったので開いてみたら・・・

switch($mode){
case "a":
case "b":
case "c":
	switch($mode){
	case "a":
		$type = "A";   
		break;
	case "b":	
		$type = "B";   
		break;
	case "c":	
		$type = "C"; 
		break;		
	}
	
	$data = array();
	$data['url'] = "F";
	if($mode == "c"){ 
		$data['custom']        = "G";
	}else{
		$data['custom']        = "H";
	}
	break;
case "d":
case "e":
	switch($mode){
	case "e":
		$type = "E";
		break;		
	}
	
	$data = array();
	if($mode == "d"){
		$data['url']        = "I";
	}else{
		$data['url']        = "J";
	}
　　　　
　　　　// 処理
　　　　
	if($mode == "d"){     
		$data['custom']        = "K";
	}else{
		$data['custom']        = "L";
      
	}
	break;	
default:
}
