投稿者からのアピールポイント

もはや人間が読むものではない。

 * ZWQI001 クラス
 */

    /**
     * m_F001
     */

    /**
     * コンストラクタ
     */
    public ZWQI001() {
    }

    /**
     * m_F001 を返却します。
     */
    public String get_m_F001() {
        return m_F001;
    }
