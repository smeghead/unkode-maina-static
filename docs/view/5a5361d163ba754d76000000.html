#include<stdio.h>

#define ATA "アタッ"
enum SUCCEED{ZERO=0,ONE=1,MINUS1=-1};

//お前はすでに死んでいるで有名な

{
    int *i;
    int j;
    i=&j;
    //ポインタにアドレスを渡す
    
    for(*i=0x1234567890;*i>=0;--*i) {
    //16進数の任意の数を0以下までデクリメントする
        Sleep(256);//なるべく速くアタタタタ
        printf(ATA //ビックリ！こんなでもコンパイル可能なのか
        "\n" //なるべく改行する
);
    }
    printf(" \"//お前はすでに死んでいる！！");//コメントしようと思ったんだが
    
    return ZERO;
}
