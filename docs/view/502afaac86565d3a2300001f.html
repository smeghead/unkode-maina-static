[Java] try catch大好き☆

投稿者からのアピールポイント

自称Java開発経験者が書いたコードです。本当に開発していたのでしょうか。。。 ・どのExceptionのログも出さないの？ ・resultは何のためにあるの？

	// 定数
	public static final String HOGEHOGE="HOGEHOGE";
	public static final String FUGAFUGA="FUGAFUGA";

	public static String judgeStr(String[] str) {

		String result=HOGEHOGE;

		try {
			if (isHoge(str)) {
				result=HOGEHOGE;
			}
		} catch (Exception e) {
			return HOGEHOGE;
		}

		try {
			if (isHuga(str)) {
				result=FUGAFUGA;
			}
		} catch (Exception e) {
			return HOGEHOGE;
		}

/* 各メソッド毎にtry catchが*/

		try {
			/* 何らかの処理 */
		} catch (Exception e)
		{
			return HOGEHOGE;
		}
		
		return result;
	}
