    Try
        Dim sb As New System.Text.StringBuilder()
        Dim capCd As String

        If capCode = "" Then
            capCd = ""
        Else
            capCd = capCode(0)
        End If

        If colorCode <> String.Empty Then
            sb.Append(colorCode)
            sb.Append("%")
            colorCode = sb.ToString()
        End If

        '中略～capCdとcolorCodeをキーにDBアクセスして結果をReturnする処理～

    Catch ex As Exception
        Throw ex
    End Try

End Function
