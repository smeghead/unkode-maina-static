ムカデ人間はクソ映画

public abstract class AbstractActivity extends Activity {
    /* うん */
}

public abstract class AbstractBaseActivity extends AbstractActivity {
    /* ん？ */
}

public class MainActivity extends AbstractBaseActivity {
    /* まあいいか・・・ */
}

public class InitializeActivity extends MainActivity {
    /* え？ */
}

public class MakeActivity extends InitializeActivity {
    /* なんだこれ */
}

public class SNSMainActivity extends MakeActivity {
    /* なんで動くんだこれ */
}

public class WebAcitivity extends SNSMainActivity {
    /* どのページだこれ */
}

public class ReserveActivity extends WebActivity {
    /* もうちょっとだけ続くんじゃ */
}

public class OptionActivity extends ReserveActivity {
    /* ・・・ */
}

public class SNSSubActivity extends OptionActivity {
    /* さっき見・・・ちょっと名前違う！ */
}

public class SNSThirdActivity extends AbstractBaseActivity {
    /* これでラスt・・・分岐しているだと？ */
}
