[Java] フルHD推奨。

投稿者からのアピールポイント

画面の左側がわりと無駄な気がします。

public int hoge(){
    int result = 0;

    try {
        if ( proc1() == 0 ){
            // 正常処理
            try {
                if ( proc2() == 0 ){
                    // 正常処理
                    try {
                        if ( proc3() == 0 ){
                            // 正常処理
                            try {
                                if ( proc4() == 0 ){
                                    // 正常処理
                                    try {
                                        if ( proc5() == 0 ){
                                            // 正常処理
                                            if ( proc6() == 0 ){
                                                // 正常処理
                                                // まだまだ続くよ！
                                            } else {
                                                // 異常処理
                                            }
                                        } else {
                                            // 異常処理
                                        }
                                    } catch(Exception e) {
                                        // 例外処理
                                    }
                                } else {
                                    // 異常処理
                                }
                            } catch(Exception e) {
                                // 例外処理
                            }
                        } else {
                            // 異常処理
                        }
                    } catch(Exception e) {
                        // 例外処理
                    }
                } else {
                    // 異常処理
                }
            } catch(Exception e) {
                // 例外処理
            }
        } else {
            // 異常処理
        }

    } catch(Exception e) {
    // 握りつぶす
    }

    return result;
}
