 */

    int bufferSize = size * flag ? 4 : 1;
    char* cBuffer = new char[bufferSize];

    /* 一度配列をコピー */
    /* TODO : メモリアクセス違反エラーが発生中。 */
    for (int i = 0; i < bufferSize; i++)
    {
        cBuffer[i] = buffer[i];
    }

    /* 以下、cBufferを使った処理 */
}
