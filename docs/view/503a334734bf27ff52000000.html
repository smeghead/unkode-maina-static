眠気のある時にコード書くと危ない。何しでかすか本当にわからん。

function next();{ //セミコロン要らねえ
  var a = 10;
  var b = a++; //何がしたいの
  var word = array(); //配列作ってんじゃなくてなんか関数代入してるし中身書いてないしそもそも定義してないし
  var next = word[b]; //関数名と被ってるし

}
} //エラー起こす原因の一つだけど時既に遅し
