[C#] 一行に詰め込まれた無駄な処理

投稿者からのアピールポイント

int unk = 0;って書けば良い処理。

Constantsクラスは実際の変数名、プロパティ名に合わせています。

public class Constants
{
    private const int ZERO = 0;
    public static int Zero
    {
        get{ return ZERO ;}
    }
}

public class Unk
{
    public Unk()
    {
        int unk = int.Parse(Constants.Zero.ToString());
    }
}
